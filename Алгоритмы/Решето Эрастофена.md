func ErastofenSieve(n int) []int {  
    composites := make([]bool, n+1)  
    if n < 2 {  
       return []int{}  
    }  
    for p := 2; p*p <= n; p++ {  
       if composites[p] == true {  
          continue  
       }  
       for i := p * p; i <= n; i += p {  
          composites[i] = true  
       }  
    }  
  
    var result []int = make([]int, 0, n)  
    for ind, val := range composites {  
       if val != true {  
          result = append(result, ind)  
       }  
    }  
    result = result[2:]  
    return result  
}
